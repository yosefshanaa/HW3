\input{preamble}

\begin{document}

% --- Cover (FR-B1) ----------------------------------------------------------
% Front matter (cover + TOC) is numbered with roman numerals (i, ii, …); the
% body restarts at arabic 1 below. This is the standard book convention and,
% crucially, keeps the cover OUT of the arabic sequence: numbering roman *before*
% the cover means the cover is page "i" (not printed), so it no longer collides
% with the first chapter as a second arabic "1" in the PDF page labels.
\pagenumbering{roman}
\input{cover}

% --- Table of contents (FR-B2) ----------------------------------------------
% The titlepage environment resets the page counter to 1 on exit, so restore it
% to 2 to keep the roman sequence continuous (cover i -> TOC ii, iii …).
\setcounter{page}{2}
\tableofcontents
\clearpage

% --- Chapters ---------------------------------------------------------------
\pagenumbering{arabic}
\input{chapters/01-intro}
\input{chapters/02-idea}

\input{chapters/04-pmf}

\input{chapters/06c-metrics}  % AARRR funnel (Python graph) + North-Star table
\input{chapters/07-pitfalls}
\input{chapters/08-summary}

% --- Bibliography (FR-B9) ---------------------------------------------------
% All sources are English-language, so typeset the reference list left-to-right.
\begin{otherlanguage}{english}
\printbibliography[heading=bibintoc]
\end{otherlanguage}

% --- Colophon (how the book was produced — CrewAI credit) -------------------
\input{colophon}
