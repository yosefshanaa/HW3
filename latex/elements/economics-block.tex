\noindent ערך החיים של הלקוח, עלות הרכישה ומסלול המזומן מוגדרים
כך~\cite{thiel2014zero}:
\begin{equation}
  \mathrm{LTV} \;=\; \frac{\overline{\mathrm{ARPA}} \cdot m_{\text{gross}}}{c_{\text{churn}}},
  \label{eq:gen-ltv}
\end{equation}
\begin{align}
  \mathrm{CAC} \;=\; \frac{S_{\text{S\&M}}}{N_{\text{new}}},
  \qquad
  \mathrm{Runway} \;=\; \frac{C_{\text{cash}}}{B_{\text{net}}}\,.
  \label{eq:gen-cac}
\end{align}
כלל האצבע לבריאות עסקית:
\begin{equation}
  \boxed{\;\dfrac{\mathrm{LTV}}{\mathrm{CAC}} \;\geq\; 3
  \quad\text{וגם}\quad \text{זמן החזר ה-}\mathrm{CAC} \;\leq\; 12\ \text{חודשים}.\;}
\end{equation}
